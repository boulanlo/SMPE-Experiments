library("ggplot2")
library("dplyr")
data <- read.delim("data.txt", header=TRUE, sep=" ")
ggplot(data = data, aes(x=year, y=pop)) +
  geom_line() +
  geom_point() +
  geom_text(aes(label=pop), vjust=-1) +
  theme_minimal() +
  ggtitle("Evolution of the population of La Clusaz (74)") +
  xlab("Year") +
  ylab("Population") +
  theme(plot.title = element_text(hjust=0.5))
